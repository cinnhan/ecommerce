public function request(string $method, string $uri): array
{
    return compact('method', 'uri');
}

private function getFunctionMethod($function): array
{
    $requestFunction = 'request';
    $method = str_replace($requestFunction, '', $function);

    return [$requestFunction, $method];
}

public function requestGet(string $uri): array
{
    list($requestFunction, $method) = $this->getFunctionMethod(__FUNCTION__);

    return $this->$requestFunction($method, $uri);
}

public function requestPost(string $uri): array
{
    list($requestFunction, $method) = $this->getFunctionMethod(__FUNCTION__);

    return $this->$requestFunction($method, $uri);
}
